  Yi = function(L,Kcorr) {
  N = length(Kcorr)
  conc = c()
  conc[1] = L*Kcorr[1]
  for (i in 2:N) conc[i] = conc[i-1]*L*Kcorr[i]
  numer2 = sum((1:N)*conc)/N
  denom2 = 1 + sum(conc)
  return(numer2/denom2)
}
